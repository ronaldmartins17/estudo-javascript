Código validar Nota

código para validar uma nota entre 0 e 10 usando if.

Parte html do código

 
    <input type=number id="notaInput" placeholder="Digite uma nota entre 0 e 10">
    <button onclick="validarNota()">Validar Nota </button>
                
    <p id="res">
        Resultado aqui !!!
    </p>

Parte javascript do código

 
    <script>

    function validarNota() {
        const notaInput = document.querySelector("#notaInput");
        let nota = Number(notaInput.value);
        const resultado = document.querySelector("#res");
        if (nota < 0 || nota > 10 || isNaN(nota)){ 
        //resultado.innerText = `nota inválida! digite uma nota entre 0 e 10: `; 
            resultado.innerText = `digite nvamente uma nota válida entre 0 e 10.`;
            notaInput.value = ''; // Limpa o campo de entrada
            notaInput.focus(); // Foca no campo novamente
            return false;
        }else{
            resultado.innerText = `nota atual: ${nota}`;
        }
    }
    </script>
